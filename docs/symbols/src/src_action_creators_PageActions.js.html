 10 /**
 11  * checks the URL query string
 12  * @param {Object} URLSearchParams object
 13  * @returns {String} indicating if parameters are valid
 14  */
 15 export const checkURL = (urlParams) => {
 16   
 17   if(typeof(urlParams) !== 'object') {
 18 
 19     //not readable
 20     return 'unreadable';
 21   } else if(urlParams.has('error')) {
 22 
 23     //has error
 24     const errMes = urlParams.get('error');
 25     return String(errMes);
 26   } else {
 27 
 28     //is valid
 29     return 'valid';
 30   }
 31 };

 33 /**
 34  * checks a valid URL query string for a magic link and updates the store accordingly
 35  * @param {Object} URLSearchParams object
 36  * @param {String} indicating if there was a valid magic link
 37 */
 38 export const confirmLink = (urlParams) => (
 39   (dispatch) => {
 40 
 41     //obtain the magic link
 42     const magicLink = urlParams.get('magiclink');
 43     if(!magicLink) {
 44 
 45       //not a valid magic link
 46 
 47       //clear link in store
 48       dispatch({
 49         type: LOGIN_MNGMNT.SET_MAGIC_LINK,
 50         magicLink: ''
 51       });
 52 
 53       //notify store
 54       dispatch({
 55         type: LOGIN_MNGMNT.HAS_FORGOTTEN_PASSWORD,
 56         forgottenPassword: false
 57       });
 58 
 59       //notify a failure
 60       return 'no magic link found';
 61     } else {
 62 
 63       //user forgot password and is attempting to reset
 64 
 65       //put the link in the store
 66       dispatch({
 67         type: LOGIN_MNGMNT.SET_MAGIC_LINK,
 68         magicLink: magicLink
 69       });
 70 
 71       //notify store
 72       dispatch({
 73         type: LOGIN_MNGMNT.HAS_FORGOTTEN_PASSWORD,
 74         forgottenPassword: true
 75       });
 76 
 77       //notify a success
 78       return 'magic link success';
 79     }
 80   }
 81 );

 83 /**
 84  * checks for cookies of a specified name
 85  * @param {String} name of the cookie
 86  * @returns {Object} cookie data || null
 87  */
 88 export const retrieveCookie = (name) => (
 89   (dispatch) => {
 90 
 91     //get the cookie
 92     const cookie = dispatch(getCookie(name));
 93 
 94     if(cookie && typeof(cookie) === 'string') {
 95 
 96       //useable cookie
 97       return JSON.parse(cookie);
 98     } else {
 99 
100       //fails
101       return null;
102     }
103   }
104 );
105 
106 /**
107  * checks the valid date of a cookie
108  * @param {Object} specified cookie
109  * @returns {Boolean} if the cookie is expired or not
110  */
111 export const checkCookieValid = (cookie) => {
112 
113   if(!cookie || !cookie.auth) {
114     
115     //bad cookie
116     return false;
117   }
118 
119   //check the validation date
120   if(cookie.auth.valid_until && Date.parse(cookie.auth.valid_until) > Date.now()) {
121 
122     //still valid
123     return true;
124   } else {
125 
126     //invalid
127     return false;
128   }
129 };

131 /**
132  * saves specified data to a specified cookie 
133  * @param {String} data to save 
134  * @param {String} name of the cookie
135  * @returns {Boolean} if save successful
136  */
137 export const saveCookie = (data, name) => (
138   (dispatch) => {
139 
140     //check if valid
141     if(typeof(data) !== 'string' || typeof(name) !== 'string') {
142 
143       //invalid
144       return false;
145     } else {
146 
147       //turn the data into an object
148       const cookie = JSON.parse(data);
149 
150       //set the authdata cookie
151       dispatch(setCookie(name, cookie));
152 
153       return true;
154     }
155   }
156 );

158 /**
159  * removes a specified cookie
160  * @param {String} name of the cookie
161  * returns {Boolean} if removal successful
162  */
163 export const deleteCookie = (name) => (
164   (dispatch) => {
165   
166     //check if string
167     if(typeof(name) !== 'string') {
168       
169       //invalid
170       return false;
171     }
172 
173 
174     //check that cookie exists
175     const cookie = dispatch(retrieveCookie(name));
176 
177     if(!cookie) {
178 
179       //non-existent
180       return false;
181     } else {
182 
183       //able to remove
184       dispatch(removeCookie(name));
185       return true;
186     }
187   }
188 );
